# Если при загрузке появляется строки командной строки с сообщением нажмите Control D или Enter

# Комбинация Control D переводит вас в режим восстановления
# Комбинация Enter в режим командной строки в консоль от имени суперпользователя

# Нажимаем комбинацию Control D  переход в режим восстановления 
# Из появившегося списка выбрать первое (верхнее) ядро с надписью (recovery mode).
	*Ubuntu, c Linux 5.3.0-53-generic (recovery mode)
# Или так, зависит какая версия ядра установлена последней
	*Ubuntu, c Linux 4.15.0-42-generic (recovery mode)

# После выбора pагрузится меню режим восстановления ubuntu в котором вы можете выбрать один из доступных вариантов:
resume - продолжить нормальную загрузку системы;
	*clean - попытаться освободить место на диске;
	*dpkg - восстановление поврежденных пакетов;
failsafeX - запустить графический безопасный режим;
	*fsck - проверить все файловые системы на ошибки;
grub - обновить настройки загрузчика Grub;
network - Включить поддержку сети;
	*root - войти в консоль от имени суперпользователя;
system-summary - информация о системе.

# Выбираем поочередно, те пункты которые отмечены звёздочкой *

# Далее после выполнения последнего пункта *root, когда вы перешли в консоль от имени суперпользователя 
вводим строки, чтобы примонтировать загрузочную систему и диски. 
и проверяем всё ли в порядке (если выдаст ошибку, например не найден UUID=XXXX-XXXX значит не всё).
# Если вы хотите вносить какие-либо правки, кроме проверки диска на ошибки в fsck, то вам придется сделать файловую систему доступной для записи

~# mount -n -o remount,rw /
~# mount -a
~# 

# Если нет ошибки значит вводими опять переход в режим восстановления

~# systemctl emergency

# Выбираем
*Ubuntu, c Linux 5.3.0-53-generic (recovery mode)
# И выбираем resume - продолжить нормальную загрузку системы;

# Если выдаёт ошибку типа can't find UUID=XXXX-XXXX
# Узнаём какой UUID диска sda1
~# ls -la /dev/disk/by-uuid/
65FA-E861 -> ../../sda1

# Редактируем fstab  и проверяем правильно ли записан UUID

~# nano /etc/fstab

# Если нет ошибка исправлена переход в режим восстановления

~# systemctl emergency

# Выбираем
*Ubuntu, c Linux 5.3.0-53-generic (recovery mode)
# И выбираем resume - продолжить нормальную загрузку системы;

# Или просто перезагрузить систему
~#  reboot

# Если вы хотите обновлять систему, то вам понадобиться интернет. Его лучше настроить на предыдущем этапе. 
Но если вы этого не сделали, то NetworkManager можно запустить и сейчас для этого выполните:
 NetworkManager

 echo "nameserver 8.8.8.8" > /etc/resolv.conf

# Аналогично, чтобы переключиться с аварийного режима в режим восстановления, введите команду:
